        // Renderizar aulas no editor
        function renderLessons() {
            const grid = document.getElementById('lessons-grid-editor');
            grid.innerHTML = lessons.map(lesson => `

                    <div class="edit-controls">

                    </div>
                    <div class="lesson-thumb">
                        ${getIconForModality(lesson.modality)}
                    </div>
                    <h4 style="color: var(--gold); margin-bottom: 0.5rem;">${lesson.title}</h4>
                    <p style="font-size: 0.9rem; opacity: 0.8;">${lesson.modality} • ${lesson.level}</p>
                    <p style="font-size: 0.8rem; opacity: 0.6; margin-top: 0.5rem;">⏱️ ${lesson.duration}</p>

        function getIconForModality(modality) {
            const icons = {
                'Samba': '💃', 'Tango': '🕺', 'Valsa': '🎵', 
                'Forró': '🔥', 'Salsa': '🌴', 'Zouk': '⚡'
            };
            return icons[modality] || '🎭';

        // Editar aula existente
        function editLesson(id) {
            const lesson = lessons.find(l => l.id === id);
            if (!lesson) return;
            
            document.getElementById('lesson-modal').classList.add('active');
            document.getElementById('modal-title').textContent = 'Editar Aula';
            document.getElementById('lesson-title').value = lesson.title;
            document.getElementById('lesson-modality').value = lesson.modality;
            document.getElementById('lesson-level').value = lesson.level;
            document.getElementById('lesson-duration').value = lesson.duration;
            
            // Guardar ID para update
            document.getElementById('lesson-modal').setAttribute('data-editing-id', id);
        }

        // Salvar aula (add ou edit)
        function saveLesson() {
            const title = document.getElementById('lesson-title').value;
            const modality = document.getElementById('lesson-modality').value;
            const level = document.getElementById('lesson-level').value;
            const duration = document.getElementById('lesson-duration').value;
            const desc = document.getElementById('lesson-desc').value;
            
            if (!title) {
                alert('Por favor, preencha o título da aula');
                return;
            }
            
            const editingId = document.getElementById('lesson-modal').getAttribute('data-editing-id');
            
            if (editingId) {
                // Update
                const index = lessons.findIndex(l => l.id == editingId);
                if (index !== -1) {
                    lessons[index] = { ...lessons[index], title, modality, level, duration };
                }
                document.getElementById('lesson-modal').removeAttribute('data-editing-id');
            } else {
                // Add new
                const newLesson = {
                    id: Date.now(),
                    title,
                    modality,
                    level,
                    duration: duration || '20:00',
                    progress: 0
                };
                lessons.push(newLesson);
            }
            
            renderLessons();
            closeModal();
            showToast(editingId ? 'Aula atualizada!' : 'Aula adicionada com sucesso!');
        }

        // Deletar aula
        function deleteLesson(id) {
            if (confirm('Tem certeza que deseja remover esta aula?')) {
                lessons = lessons.filter(l => l.id !== id);
                renderLessons();
                showToast('Aula removida');
            }
        }

        // Setup listeners para edição inline
        function setupEditableListeners() {
            document.querySelectorAll('.editable-text').forEach(el => {
                el.addEventListener('click', function(e) {
                    if (isPreview) return;
                    e.stopPropagation();
                    
                    // Cria input temporário para edição inline
                    const currentText = this.textContent;
                    const input = document.createElement('input');
                    input.type = 'text';
                    input.value = currentText;
                    input.style.cssText = `
                        background: rgba(0,0,0,0.8);
                        color: var(--gold);
                        border: 2px solid var(--gold);
                        padding: 0.5rem;
                        font-size: inherit;
                        font-family: inherit;
                        width: 100%;
                        border-radius: 4px;
                    `;
                    
                    this.style.display = 'none';
                    this.parentNode.insertBefore(input, this.nextSibling);
                    input.focus();
                    input.select();
                    
                    const saveEdit = () => {
                        this.textContent = input.value || currentText;
                        this.style.display = '';
                        input.remove();
                        showToast('Texto atualizado');
                    };
                    
                    input.addEventListener('blur', saveEdit);
                    input.addEventListener('keypress', (e) => {
                        if (e.key === 'Enter') saveEdit();
                    });
                });
            });
        }
